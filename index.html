import pygame import random # Инициализация Pygame pygame.init() # Настройки экрана WIDTH, HEIGHT = 600, 400 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("LEGO Minifigure Pac-Man Game") # Цвета BLACK = (0, 0, 0) WHITE = (255, 255, 255) GRAY = (200, 200, 200) # Игрок (Pac-Man) class LegoPacMan: def __init__(self): self.image = pygame.image.load('lego_pacman.png') # Замените на путь к вашему спрайту Pac-Man self.rect = self.image.get_rect(topleft=(50, 50)) self.speed = 5 def move(self, dx, dy): self.rect.x += dx * self.speed self.rect.y += dy * self.speed def draw(self): screen.blit(self.image, self.rect) # Отрисовка лабиринта def draw_maze(): for x in range(0, WIDTH, 40): for y in range(0, HEIGHT, 40): if (x // 40 + y // 40) % 2 == 0: pygame.draw.rect(screen, GRAY, (x, y, 40, 40)) # Основной игровой цикл def main(): clock = pygame.time.Clock() pacman = LegoPacMan() running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False keys = pygame.key.get_pressed() dx = keys[pygame.K_RIGHT] - keys[pygame.K_LEFT] dy = keys[pygame.K_DOWN] - keys[pygame.K_UP] # Обработка границ экрана pacman.move(dx, dy) if pacman.rect.x < 0: pacman.rect.x = 0 if pacman.rect.x > WIDTH - pacman.rect.width: pacman.rect.x = WIDTH - pacman.rect.width if pacman.rect.y < 0: pacman.rect.y = 0 if pacman.rect.y > HEIGHT - pacman.rect.height: pacman.rect.y = HEIGHT - pacman.rect.height # Отрисовка screen.fill(BLACK) draw_maze() pacman.draw() pygame.display.flip() clock.tick(30) pygame.quit() if __name__ == "__main__": main()
